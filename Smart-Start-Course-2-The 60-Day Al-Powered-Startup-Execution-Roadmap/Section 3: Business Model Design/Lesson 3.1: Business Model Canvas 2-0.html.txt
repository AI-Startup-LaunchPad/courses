        // Canvas management
        function updateCanvasBlock(component, value) {
            canvasData[component] = value;
            const block = document.querySelector(`[data-component="${component}"]`);
            
            if (value.trim().length > 10) {
                block.classList.add('completed');
                if (!block.dataset.completed) {
                    block.dataset.completed = 'true';

                if (block.dataset.completed) {
                    block.dataset.completed = 'false';
                    completedComponents--;
                    updateProgress();
                }
            }
            
            saveProgress();
        }

            const components = ['partners', 'activities', 'resources', 'value', 'relationships', 'channels', 'segments', 'costs', 'revenue'];

        // AI Assistant
        function processAIPrompt() {
            const prompt = document.getElementById('aiPrompt').value;
            const responseDiv = document.getElementById('aiResponse');
            
            if (!prompt.trim()) return;
            
            // Simulate AI response
            responseDiv.style.display = 'block';
            responseDiv.innerHTML = '<div style="color: var(--slate-600); font-style: italic;">AI is analyzing your question...</div>';

                const responses = [
                    "For your value proposition, focus on the specific pain point you're solving and quantify the benefit. Consider using the formula: 'We help [target customer] achieve [desired outcome] by [unique approach].'",

            }, 1500);
            
            document.getElementById('aiPrompt').value = '';
        }

        // Checklist management
        function toggleChecklistItem(item) {
            const checkbox = item.querySelector('.checkbox');
            const isChecked = checkbox.classList.contains('checked');
            
            if (isChecked) {
                checkbox.classList.remove('checked');
                item.classList.remove('completed');
            } else {
                checkbox.classList.add('checked');
                item.classList.add('completed');
            }
            
            updateProgress();
            saveProgress();
        }

        // Progress management
        function updateProgress() {
            const canvasProgress = (completedComponents / 9) * 30;
            const checklistItems = document.querySelectorAll('.checklist-item.completed').length;
            const checklistProgress = (checklistItems / 5) * 20;
            
            currentProgress = 45 + canvasProgress + checklistProgress;
            
            const progressFill = document.querySelector('.progress-fill');
            const progressLabel = document.querySelector('.progress-label span:last-child');
            
            progressFill.style.width = `${currentProgress}%`;
            progressLabel.textContent = `${Math.round(currentProgress)}% Complete`;
        }

        function saveProgress() {
            const progressData = {
                currentProgress,
                canvasData,
                completedComponents,
                checklistStates: Array.from(document.querySelectorAll('.checklist-item')).map(item => 
                    item.classList.contains('completed')
                )
            };
            localStorage.setItem('smartstart_lesson_3_1', JSON.stringify(progressData));
        }

        function loadProgress() {
            const saved = localStorage.getItem('smartstart_lesson_3_1');
            if (saved) {
                const data = JSON.parse(saved);
                currentProgress = data.currentProgress || 45;
                canvasData = data.canvasData || {};
                completedComponents = data.completedComponents || 0;
                
                // Restore checklist states
                if (data.checklistStates) {
                    const checklistItems = document.querySelectorAll('.checklist-item');
                    data.checklistStates.forEach((isCompleted, index) => {
                        if (isCompleted && checklistItems[index]) {
                            checklistItems[index].classList.add('completed');
                            checklistItems[index].querySelector('.checkbox').classList.add('checked');
                        }
                    });
                }
                
                updateProgress();
            }
        }

        function completeLesson() {
